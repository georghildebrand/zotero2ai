```python
"""Command-line interface for zotero2ai."""

import argparse
from typing import Sequence


def parse_args(args: Sequence[str] | None = None) -> argparse.Namespace:
    """
    Parse command-line arguments.

    Args:
        args: Arguments to parse (defaults to sys.argv[1:])

    Returns:
        Parsed arguments namespace
    """
    parser = argparse.ArgumentParser(
        prog="mcp-zotero2ai",
        description="MCP server for Zotero with hybrid search",
    )

    # Global flags
    parser.add_argument(
        "--debug",
        action="store_true",
        help="Enable debug logging",
    )
    parser.add_argument(
        "--quiet",
        action="store_true",
        help="Only show warnings and errors",
    )

    # Subcommands
    subparsers = parser.add_subparsers(dest="command", required=True)

    # Doctor command
    subparsers.add_parser(
        "doctor",
        help="Validate Zotero installation and configuration",
    )

    # Run command with transport modes
    run_parser = subparsers.add_parser(
        "run",
        help="Start MCP server",
    )

    # Transport mode (mutually exclusive group)
    transport_group = run_parser.add_mutually_exclusive_group()
    transport_group.add_argument(
        "--stdio",
        action="store_true",
        default=True,
        help="Use stdio transport (default)",
    )
    transport_group.add_argument(
        "--http",
        action="store_true",
        help="Use HTTP transport for ChatGPT integration",
    )

    # HTTP-specific options
    run_parser.add_argument(
        "--host",
        type=str,
        default="127.0.0.1",
        help="HTTP server host (default: 127.0.0.1)",
    )
    run_parser.add_argument(
        "--port",
        type=int,
        default=8787,
        help="HTTP server port (default: 8787, 0 for ephemeral)",
    )
    run_parser.add_argument(
        "--once",
        action="store_true",
        help="Start server, bind, and exit immediately (smoke test)",
    )

    # Parse and normalize transport flags
    parsed_args = parser.parse_args(args)

    # Normalize: if --http is set, stdio should be False
    if hasattr(parsed_args, "http") and parsed_args.http:
        parsed_args.stdio = False

    return parsed_args
```

```python
"""Command-line interface for zotero2ai."""

import argparse
import logging
import sqlite3
from typing import Sequence

from zotero2ai.config import ZoteroConfig

logger = logging.getLogger(__name__)


def parse_args(args: Sequence[str] | None = None) -> argparse.Namespace:
    """
    Parse command-line arguments.

    Args:
        args: Arguments to parse (defaults to sys.argv[1:])

    Returns:
        Parsed arguments namespace
    """
    parser = argparse.ArgumentParser(
        prog="mcp-zotero2ai",
        description="MCP server for Zotero with hybrid search",
    )

    # Global flags
    parser.add_argument(
        "--debug",
        action="store_true",
        help="Enable debug logging",
    )
    parser.add_argument(
        "--quiet",
        action="store_true",
        help="Only show warnings and errors",
    )

    # Subcommands
    subparsers = parser.add_subparsers(dest="command", required=True)

    # Doctor command
    subparsers.add_parser(
        "doctor",
        help="Validate Zotero installation and configuration",
    )

    # Run command
    subparsers.add_parser(
        "run",
        help="Start MCP server",
    )

    return parser.parse_args(args)


def cmd_doctor(config: ZoteroConfig) -> int:
    """
    Run doctor command to validate Zotero setup.

    Args:
        config: Zotero configuration

    Returns:
        Exit code (0 = success, 1 = error)
    """
    print("🔍 zotero2ai Doctor")
    print()
    print(f"✓ Zotero data directory: {config.data_dir}")
    print(f"✓ Database: {config.db_path}")
    print(f"✓ Storage: {config.storage_path}")
    print()

    # Basic SQLite sanity check
    try:
        conn = sqlite3.connect(f"file:{config.db_path}?mode=ro", uri=True)
        cursor = conn.cursor()
        cursor.execute("SELECT COUNT(*) FROM items")
        item_count = cursor.fetchone()[0]
        conn.close()
        print(f"✓ Database accessible: {item_count} items")
    except Exception as e:
        print(f"✗ Database error: {e}")
        logger.error(f"Database validation failed: {e}")
        return 1

    print()
    print("All checks passed! 🎉")
    return 0
```

```python
def cmd_run(config: ZoteroConfig, stdio: bool, http: bool, host: str, port: int, once: bool) -> int:
    """
    Run MCP server with stdio or HTTP transport.

    Creates FastMCP server with one trivial tool (ping) to validate integration.
    Supports stdio (default) and HTTP modes (for ChatGPT integration via tunnel).

    Args:
        config: Zotero configuration
        stdio: Use stdio transport
        http: Use HTTP transport
        host: HTTP server host
        port: HTTP server port (0 for ephemeral)
        once: Start, bind, and exit immediately (smoke test)

    Returns:
        Exit code (0 = success)
    """
    from mcp.server.fastmcp import FastMCP

    transport = "stdio" if stdio else "HTTP"
    logger.info(f"Starting MCP server with {transport} transport, data dir: {config.data_dir}")

    # Create minimal FastMCP server
    mcp = FastMCP("zotero2ai")

    @mcp.tool()
    def ping() -> str:
        """Ping the server to check if it's alive."""
        return "pong"

    if stdio:
        print("🚀 MCP server started (0.1.0 stub with ping tool)")
        print("Transport: stdio")
        print()
        print("Available tools:")
        print("  - ping: Check if server is alive")
        print()
        logger.debug("MCP server initialized successfully (stdio mode)")
        # For 0.1.0, stdio mode is a stub that validates setup and exits
        return 0

    elif http:
        print(f"🚀 MCP server started (HTTP mode)")
        print(f"Host: {host}")
        print(f"Port: {port if port != 0 else '[ephemeral]'}")
        print()
        print("Available tools:")
        print("  - ping: Check if server is alive")
        print()

        if once:
            # Smoke test: validate HTTP setup and exit immediately
            logger.debug(f"HTTP mode smoke test: --once flag set, exiting without binding")
            print("✓ HTTP mode validated (--once flag: not binding)")
            return 0
        else:
            # For 0.1.0, HTTP mode without --once is also a stub
            # In v0.2.0+, this will actually call mcp.run() with HTTP transport
            logger.debug("HTTP server stub initialized (full HTTP server in v0.2.0+)")
            print("Note: Full HTTP server implementation coming in v0.2.0")
            return 0

    return 0
```

```python
def main() -> int:
    """
    Main entry point for CLI.

    Returns:
        Exit code
    """
    import sys

    from zotero2ai.config import resolve_zotero_data_dir
    from zotero2ai.logging import setup_logging

    try:
        # Parse arguments
        args = parse_args()

        # Setup logging
        setup_logging(debug=args.debug, quiet=args.quiet)

        # Resolve configuration
        data_dir = resolve_zotero_data_dir()
        config = ZoteroConfig(data_dir=data_dir)

        # Dispatch command
        if args.command == "doctor":
            return cmd_doctor(config)
        elif args.command == "run":
            # Pass transport mode arguments to cmd_run
            return cmd_run(
                config,
                stdio=args.stdio,
                http=args.http,
                host=args.host,
                port=args.port,
                once=args.once,
            )
        else:
            logger.error(f"Unknown command: {args.command}")
            return 1

    except ValueError as e:
        # Configuration error (missing ZOTERO_DATA_DIR)
        print(f"Configuration error: {e}", file=sys.stderr)
        return 1
    except FileNotFoundError as e:
        # Invalid Zotero directory
        print(f"Error: {e}", file=sys.stderr)
        return 1
    except KeyboardInterrupt:
        logger.info("Interrupted")
        return 130
    except Exception as e:
        logger.exception(f"Unexpected error: {e}")
        return 1


if __name__ == "__main__":
    import sys
    sys.exit(main())
```
